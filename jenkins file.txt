pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main' , url: 'https://github.com/vishal-user/TravelMemory.git'
            }
        }
        stage('install') {
            steps {
                 sh 'cd backend; npm install'
            }
        }
        stage('Build') {
            steps {
                 sh 'cd backend; npm run build'
            }
        }
    }
}
